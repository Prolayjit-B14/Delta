By Arghya Roy

üìÖ Post Date: February 15, 2025

Concept

The Smart Agriculture Monitoring System is an IoT-based project that integrates Arduino Uno with various sensors and components to monitor and control agricultural conditions in real time. The system collects environmental data, processes it, and provides alerts or automated actions to optimize farming operations.

Components Required

  • Arduino Uno ‚Äì The microcontroller that processes sensor data and controls components.
  • LED Indicators (Red, Green, Blue) ‚Äì Indicate different system states.
  • Temperature Sensor (TMP36) ‚Äì Measures the surrounding temperature for real-time monitoring.
  • Buzzer ‚Äì Provides audible alerts for critical conditions.
  • LCD Display ‚Äì Shows real-time sensor readings and status messages.
  • Relay Module (SPDT) ‚Äì Controls a DC motor for automation.
  • DC Motor ‚Äì Simulates an irrigation pump or any mechanical farming task.
  • Function Generator ‚Äì Provides signals to test and control the motor.
  • IoT Integration ‚Äì Sends data to the cloud for remote monitoring via a web or mobile app.

Connections

  • Temperature Sensor ‚Üí Arduino Analog Pin A0
  • Buzzer ‚Üí Arduino Digital Pin 9
  • LCD Display ‚Üí Arduino Digital Pins 7, 8, 9, 10, 11, 12
  • Relay Module ‚Üí Arduino Digital Pin 6
  • DC Motor ‚Üí Controlled via the Relay Module
  • LED Indicators ‚Üí Arduino Digital Pins 3, 4, 5

Arduino Code


// C++ code

#include 
#include 

// Define pin connections
const int sensorPin = A0;   // Moisture sensor data pin
const int sensorPower = 8;  // Moisture sensor power control
const int tmpPin = A1;      // TMP36 Temperature Sensor
const int LED1 = 2;         // Red LED (Dry)
const int LED2 = 3;         // Yellow LED (Moderate)
const int LED3 = 4;         // Green LED (Wet)
const int buzzer = 7;       // Buzzer for dry alert
const int relayPin = 5;     // Relay for water pump
const int functionGen = 6;  // Function Generator (PWM Output)

// Initialize LCD (I2C Address 0x27)
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Moisture Sensor Thresholds
const int dryThreshold = 500;   // Below this ‚Üí Dry
const int wetThreshold = 800;   // Above this ‚Üí Wet

// Buzzer frequencies for a good sound
const int dryBuzzerFreq = 1000;    // Frequency in Hz (Dry)
const int wetBuzzerFreq = 2300;    // Frequency for Wet

void setup() {
  Serial.begin(9600);
  lcd.init();
  lcd.backlight();

  // Set Pin Modes
  pinMode(LED1, OUTPUT);
  pinMode(LED2, OUTPUT);
  pinMode(LED3, OUTPUT);
  pinMode(buzzer, OUTPUT);
  pinMode(relayPin, OUTPUT);
  pinMode(sensorPower, OUTPUT);
  pinMode(functionGen, OUTPUT);
}

void loop() {
  // üå°Ô∏è Read TMP36 Temperature
  int rawTemp = analogRead(tmpPin);
  float voltage = rawTemp * (5.0 / 1023.0);  // Convert to voltage
  float temperature = (voltage - 0.5) * 100;  // Convert to Celsius

  // üíß Read Soil Moisture
  digitalWrite(sensorPower, HIGH);
  delay(10);
  int soilMoisture = analogRead(sensorPin);
  digitalWrite(sensorPower, LOW);

  // üñ•Ô∏è Serial Monitor Debugging
  Serial.print("Soil Moisture: ");
  Serial.println(soilMoisture);
  Serial.print("Temperature: ");
  Serial.println(temperature);

  // üìü LCD Display Update
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Temp: ");
  lcd.print(temperature);
  lcd.print("C ");
  lcd.setCursor(0, 1);
  lcd.print("Soil: ");
  lcd.print(soilMoisture);  // Display soil moisture

  // üåæ Soil Moisture Classification & Actions
  if (soilMoisture < dryThreshold) {
    digitalWrite(LED1, HIGH);  // Dry
    digitalWrite(LED2, LOW);
    digitalWrite(LED3, LOW);
    tone(buzzer, dryBuzzerFreq, 500);  // Play dry alert tone for 500 ms
    digitalWrite(relayPin, LOW);  // Pump ON (Assuming LOW-ACTIVE Relay)
    lcd.print(" - DRY - PUMP ON!");
  } 
  else if (soilMoisture >= dryThreshold && soilMoisture < wetThreshold) {  // Moderate condition
    digitalWrite(LED1, LOW);
    digitalWrite(LED2, HIGH);  // Moderate
    digitalWrite(LED3, LOW);
    digitalWrite(buzzer, LOW);  // No buzzer for moderate
    digitalWrite(relayPin, HIGH);  // Pump OFF
    lcd.print(" - MODERATE");
  } 
  else {
    digitalWrite(LED1, LOW);
    digitalWrite(LED2, LOW);
    digitalWrite(LED3, HIGH);  // Wet
    tone(buzzer, wetBuzzerFreq, 500);  // Play wet alert tone for 500 ms
    digitalWrite(relayPin, HIGH);  // Pump OFF
    lcd.print(" - WET - NO WATER");
  }

  // üì° Function Generator Output (PWM)
  analogWrite(functionGen, map(soilMoisture, 300, 900, 0, 255));

  delay(2000);  // Delay before next reading
}
